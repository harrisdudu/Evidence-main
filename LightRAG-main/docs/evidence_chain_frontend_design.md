// src/features/EvidenceChain/hooks/useEvidenceQuery.ts
import { useState, useCallback } from 'react'
import { evidenceApi } from '@/api/evidence'
import type { EvidenceQueryParams, EvidenceResponse, EvidenceStats } from '@/types/evidence'

export function useEvidenceQuery() {
  const [queryParams, setQueryParams] = useState<EvidenceQueryParams>({
    evidence_levels: [],
    relation_types: [],
    scene_tags: [],
    keyword: '',
    page: 1,
    page_size: 20,
    sort_by: 'relevance',
    sort_order: 'desc'
  })

  const [results, setResults] = useState<EvidenceResponse | null>(null)
  const [stats, setStats] = useState<EvidenceStats | null>(null)
  const [loading, setLoading] = useState(false)
  const [error, setError] = useState<string | null>(null)

  const refetch = useCallback(async () => {
    setLoading(true)
    setError(null)
    try {
      const [res, statsRes] = await Promise.all([
        evidenceApi.query(queryParams),
        evidenceApi.getStats({ 
          scene_tag: queryParams.scene_tags?.[0] 
        })
      ])
      setResults(res.data)
      setStats(statsRes.data)
    } catch (err) {
      setError(err.message)
    } finally {
      setLoading(false)
    }
  }, [queryParams])

  return {
    queryParams,
    setQueryParams,
    results,
    stats,
    loading,
    error,
    refetch
  }
}
